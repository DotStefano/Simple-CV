% ============================================================
% ---- Macro de sección y estructura base
% ============================================================

% ---- Fila estilo "empresa a la izquierda / ubicación a la derecha" + "cargo / fechas"
% Uso: \CVRole{Cargo}{Empresa}{Ubicación}{Fechas}{\item ... \item ...}
\newcommand{\CVRole}[5]{%
  \par\addvspace{8pt}%
  % Fila 1: Empresa — Ubicación (derecha)
  \noindent\begin{tabularx}{\linewidth}{@{}X >{\raggedleft\arraybackslash}p{.36\linewidth}@{}}
    \textbf{#2} & \textit{#3} \\
  \end{tabularx}\par
  % Fila 2: Cargo (versalitas) — Fechas (cursiva, derecha)
  \noindent\begin{tabularx}{\linewidth}{@{}X >{\raggedleft\arraybackslash}p{.36\linewidth}@{}}
    \textsc{#1} & \textit{#4} \\
  \end{tabularx}\par
  % Viñetas
  \begin{CVList}
    #5
  \end{CVList}
}

% ---- Ajustes de estilo (puedes cambiarlos)
\newcommand{\CVruleheight}{0.6pt}   % grosor de la línea
\newcommand{\CVsectiongap}{0.6em}   % espacio entre el título y la línea

% ---- Título de sección con línea que parte desde el texto
\newcommand{\CVSection}[1]{%
  \par\addvspace{14pt}%
  \noindent\leavevmode
  {\large\bfseries #1}%
  \hspace{\CVsectiongap}%
  \leaders\hrule height \CVruleheight\hfill\kern0pt
  \par\addvspace{8pt}%
}

% ---- Entrada tipo (experiencia, educación, etc.)
% Uso: \CVItem{Título}{Entidad}{Ubicación}{Fechas}
\newcommand{\CVItem}[4]{%
  \noindent\textbf{#1}\\
  #2\\
  \ifthenelse{\isempty{#3}}{}{#3\\}%
  \ifthenelse{\isempty{#4}}{}{\emph{#4}}%
  \par\vspace{3pt}%
}

% ---- Lista compacta de ítems
\newenvironment{CVList}{%
  \begin{itemize}[leftmargin=1.2em,itemsep=2pt,topsep=2pt]}{\end{itemize}}
